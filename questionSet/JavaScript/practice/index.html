    var arr = [1, 2, 2, 3, 5, 5, 6, 8, 8, 9]
var i = 0

while (i < arr.length) {
  let item = arr[i]
  let p1 = arr.indexOf(item) // 从数组首部查找第一个匹配元素的位置
  let p2 = arr.lastIndexOf(item) // 从数组尾部查找第一个匹配元素的位置
  if (p1 === p2) { // 两个位置一直表示元素没有重复
    i++ // 移到下一个元素
  } else {
    arr.splice(p2, 1) // 删除后面重复的元素
  }
}

console.log(arr) // // [1,2,3,5,6,8,9]

    冒泡排序

    相邻两个元素比较, 如何前一个值大于后一个，则交换位置， 每趟中最大的元素会排在最后，第一趟需要比较 n - 1 次， 后面每趟比较的次数依次递减

    for (let i = 0; i < list.length - 1; i++) {
  for (let j = 0; j < list.length - i - 1; j++) {
    if (list[j] > list[j + 1]) {
      let tmp = list[j]
      list[j] = list[j + 1]
      list[j + 1] = tmp
    }
  }
}

    找到数组的中间值，使小于中间值的排在左边，大于中间值的排在右边，左边的任意值都要小于右边的值，然后再次对左右两边进行分割以此类推

    /**
 * 指定深度，扁平化数组
 * @param {*} arr 要展开的数组
 * @param {*} maxLayer 展开深度，默认为1
 * @param {*} curLayer
 * @returns 返回新的数组
 */
function flatternArray(arr, maxLayer = 1, curLayer = 0) {
  let newArr = []
  curLayer++
  for (let item of arr) {
    if (curLayer > maxLayer) {
      newArr.push(item)
    } else {
      if (item instanceof Array) {
        newArr = newArr.concat(flatternArray(item, maxLayer, curLayer))
      } else {
        newArr.push(item)
      }
    }
  }
  return newArr
}

// 四维数组
var arr = [[1, 2], [3, [4, [5, 6]]]]
console.log(flatternArray(arr)) // arr.flat()
console.log(flatternArray(arr, 1)) // arr.flat(1)
console.log(flatternArray(arr, 2)) // arr.flat(2)
console.log(flatternArray(arr, 3)) // arr.flat(3)

// 打印
// [ 1, 2, 3, [ 4, [ 5, 6 ] ] ]
// [ 1, 2, 3, [ 4, [ 5, 6 ] ] ]
// [ 1, 2, 3, 4, [ 5, 6 ] ]
// [ 1, 2, 3, 4, 5, 6 ]

  var newObj = {}
  for (let key in obj) {
    if (typeof obj[key] === 'object' && obj[key] !== null) {

    /**
 * 拷贝实例对象上的属性
 * @param {*} targetObj 目标对象
 * @param {*} sourceObj 源对象
 * @returns 
 */
function cloneAttrs(targetObj, sourceObj) {
  if (typeof sourceObj !== 'object' || sourceObj === null) return sourceObj
  for (let key in sourceObj) {
    let item = sourceObj[key]
    if (typeof item !== 'object' || item === null) {
      targetObj[key] = item
    } else {
      targetObj[key] = cloneAttrs({}, item)
    }
  }
  return targetObj
}

class Rectangle {
  constructor(width, height, color) {
    this.width = width
    this.height = height
    this.color = color
  }

  get area() {
    return this.getArea()
  }

  getArea() {
    return this.width * this.height
  }
}

var rect = new Rectangle(2, 3, { border: 'black', bg: 'orange' })
var cloneRect = new rect.constructor() // 创建一个新的实例对象
cloneRect = cloneAttrs(cloneRect, rect) // 拷贝源对象上的属性
cloneRect.width = 3
cloneRect.color.bg = 'green'

console.log(rect, rect.area)
console.log(cloneRect, cloneRect.area)

    function deepClone(obj, type) {
  if (typeof obj !== 'object' || obj === null) return obj

  if (type === 'array') {
    var newArr = []
    for (let item of obj) {
      if (typeof item === 'object' && item !== null) {
        if (item instanceof Array) {
          newArr.push(deepClone(item, 'array'))
        } else {
          newArr.push(deepClone(item))
        }
      } else {
        newArr.push(item)
      }
    }
    return newArr
  }

  var newObj = {}
  for (let key in obj) {
    let item = obj[key]
    if (typeof item !== 'object' || item === null) {
      newObj[key] = item
      continue
    }
    if (item instanceof Set) {
      newObj[key] = new Set(item)
    } else if (item instanceof Map) {
      newObj[key] = new Map(item)
    } else if (item instanceof RegExp) {
      newObj[key] = new RegExp(item)
    } else if (item instanceof Array) {
      newObj[key] = deepClone(item, 'array')
    } else {
      newObj[key] = deepClone(item)
    }
  }
  return newObj
}
